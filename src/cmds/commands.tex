% print notation of the a path

% param2 start of the path
% param2 goal of the path
\NewDocumentCommand{\pathList}{m m m}{%
    \IfStrEq{#1}{o}{%
    \wrapMath{path(#2, #3)}%
    }{%
    \IfStrEq{#1}{t}{%

    }{%
        \IfStrEq{#1}{p}{%
            \wrapMath{path'(#2, #3)}%
        }{%
            \errmessage{invalid character for pathList macro}%
        }%
    }%
    }%
}
