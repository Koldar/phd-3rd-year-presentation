% print the name of a base IA relation
\NewDocumentCommand{\iabrel}{m}{%
    \IfStrEq{#1}{<}{%
        \wrapMath{<}%
    }{
        \IfStrEq{#1}{>}{%
            \wrapMath{>}%
        }{
            \IfStrEq{#1}{m}{%
                \textit{meets}%
            }{
                \IfStrEq{#1}{mi}{%
                    \textit{met-by}%
                }{
                    \IfStrEq{#1}{d}{%
                        \textit{during}%
                    }{
                        \IfStrEq{#1}{di}{%
                            \textit{contains}%
                        }{
                            \IfStrEq{#1}{o}{%
                                \textit{overlaps}%
                            }{
                                \IfStrEq{#1}{oi}{%
                                    \textit{overlapped-by}%
                                }{
                                    \IfStrEq{#1}{s}{%
                                        \textit{starts}%
                                    }{
                                        \IfStrEq{#1}{si}{%
                                            \textit{started-by}%
                                        }{
                                            \IfStrEq{#1}{f}{%
                                                \textit{finishes}%
                                            }{
                                                \IfStrEq{#1}{fi}{%
                                                    \textit{finished-by}%
                                                }{
                                                    \IfStrEq{#1}{=}{%
                                                        \textit{equals}%
                                                    }{
                                                        \errmessage{invalid character for iabrel macro}%
                                                    }   
                                                }   
                                            }   
                                        }   
                                    }   
                                }   
                            }   
                        }   
                    }   
                }   
            }   
        }    
    }
}
